// Simple generic function with type inference
fn max(a: T, b: T) -> T {
    if a > b {
        pass a;
    } else {
        pass b;
    }
}

// Result type with error handling
fn divide(a: int32, b: int32) -> int32 or DivisionError {
    if b == 0 {
        fail DivisionError.DivideByZero;
    }
    pass a / b;
}

// Entry point
fn main() {
    let result = max(42, 100);      // Type inferred as int32
    print("Max: ", result);          // Prints: Max: 100
    
    let quotient = divide(10, 2)?;  // ? operator propagates errors
    print("Result: ", quotient);     // Prints: Result: 5
}
